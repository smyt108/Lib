using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Runtime.Serialization;
using System.Runtime.Serialization.Formatters.Binary;
using System.Text;
using System.Threading.Tasks;
using System.Xml.Serialization;

namespace Citi.GSM.ExcelPlugins.Utils
{
    public static class XmlSerialization
    {
        public static T XmlDeserialize<T>(string fullName)
        {
            Stream stream = null;
            T retVal = default(T);
            try
            {
                XmlSerializer xs = new XmlSerializer(typeof(T));
                stream = new FileStream(fullName, FileMode.Open, FileAccess.Read, FileShare.Read);
                retVal = (T)xs.Deserialize(stream);
            }
            catch
            {
            }
            finally
            {
                if (stream != null)
                    stream.Close();
            }

            return retVal;
        }

        public static bool XmlSerialize(string fullName, System.Type type, object target)
        {

            bool isSaved = false;


            using (StreamWriter writer = new System.IO.StreamWriter(fullName, false))
            {
                XmlSerializer serializer = new System.Xml.Serialization.XmlSerializer(type);
                serializer.Serialize(writer, target);
                isSaved = true;
            }




            return isSaved;
        }

        public static string GetExecutingAssemblyDirectoryName()
        {
            string location = System.Reflection.Assembly.GetExecutingAssembly().Location;
            return System.IO.Path.GetDirectoryName(location);
        }


    }
}
