using System;
using System.Collections.Generic;
using System.Data.OleDb;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Citi.GSM.ExcelPlugins
{
    public class DBConnection
    {
        public string conString = @"Provider=Microsoft.ACE.OLEDB.12.0;Data Source=C:\Users\yt59849.NAM\Desktop\Database.accdb";
        public OleDbConnection mycon = null;
        public OleDbDataReader myReader = null;
        public OleDbCommand myCommand = null;

        public void openConn()
        {
            mycon = new OleDbConnection(conString);
            mycon.Open();
        }

        public void closeConn()
        {
            mycon.Dispose();
            mycon.Close();
        }

        public List<InfoObject> searchInfoByGeids(List<string> geids)
        {
            List<InfoObject> infoList = new List<InfoObject>();
            foreach (string geid in geids)
            {
                string sql = "select SOEID,CNAME,ENAME from AddInTest where GEID = '{0}'";
                sql = string.Format(sql, geid);
                myCommand = new OleDbCommand(sql, mycon);

                myReader = myCommand.ExecuteReader();

                if (myReader.Read())
                {
                    InfoObject info = new InfoObject();
                    info.GeID = geid;
                    info.SoeID = myReader["SOEID"].ToString();
                    info.EName = myReader["ENAME"].ToString();
                    info.CName = myReader["CNAME"].ToString();
                    infoList.Add(info);
                }
                else
                {
                    InfoObject info = new InfoObject();
                    info.GeID = geid;
                    info.SoeID = "Null";
                    info.EName = "Null";
                    info.CName = "Null";
                    infoList.Add(info);
                }
                myReader.Close();
            }
            return infoList;
        }

        public InfoObject searchInfoByGeid(string geid)
        {

            string sql = "select SOEID,CNAME,ENAME from AddInTest where GEID = '{0}'";
            sql = string.Format(sql, geid);
            myCommand = new OleDbCommand(sql, mycon);

            myReader = myCommand.ExecuteReader();

            if (myReader.Read())
            {
                InfoObject info = new InfoObject();
                info.GeID = geid;
                info.SoeID = myReader["SOEID"].ToString();
                info.EName = myReader["ENAME"].ToString();
                info.CName = myReader["CNAME"].ToString();
                myReader.Close();
                return info;
            }
            else
            {
                InfoObject info = new InfoObject();
                info.GeID = geid;
                info.SoeID = "Null";
                info.EName = "Null";
                info.CName = "Null";
                myReader.Close();
                return info;
            }
        }
    }
}
